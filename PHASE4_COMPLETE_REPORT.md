```python
sina = SinaAShareSource()

print(f"贵州茅台：¥{quote['close']:.2f}, 涨跌：{quote['change_pct']:.2f}%")
```

### 2. 跨交易所套利
```python
from src.trading.arbitrage.cross_exchange import CrossExchangeArbitrage

arb = CrossExchangeArbitrage(
    exchanges=['binance', 'okx', 'bybit'],
    min_profit_pct=0.3
)

opps = arb.scan_opportunities(['BTC/USDT', 'ETH/USDT'])
for opp in opps:
    print(f"套利：{opp.symbol} 利润{opp.profit_pct:.2f}%")
```
